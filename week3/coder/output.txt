import time

start_time = time.time()

num_terms = 10000
series_sum = 0.0

for i in range(num_terms):
    term = (-1)**i / (2*i + 1)
    series_sum += term

result = series_sum * 4

end_time = time.time()
execution_time = end_time - start_time

print(f"Result: {result}")
print(f"Execution time: {execution_time} seconds")
